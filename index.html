'; // Функция для отправки сообщения в Telegram function sendMessageToTelegram($token, $chatId, $message) { $url = 'https://api.telegram.org/bot' . $token . '/sendMessage'; $data = [ 'chat_id' => $chatId, 'text' => $message, 'parse_mode' => 'Markdown' ]; $options = [ 'http' => [ 'header' => "Content-type: application/x-www-form-urlencoded\r\n", 'method' => 'POST', 'content' => http_build_query($data), ], ]; $context = stream_context_create($options); $result = file_get_contents($url, false, $context); return $result; } ?>
